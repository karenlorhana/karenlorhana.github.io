comandos LSA


mkdir = cria diretório
groupadd = cria grupo
groupadd -g = cria grupo add o id
groupdel = apaga grupo


/etc/passwd = lista usuários
/etc/shadow = onde é armazenado senhas
/etc/group = lista grupos


useradd = cria usuario
adduser nomeUser nomeGrupo = add user ao grupo
userdel = apaga usuario
usermod = mudar dados de um user
groupmod = mudar dados de um grupo


chown nomeUsuario:nomeGrupo nomeDiretorio = muda o adm e o grupo de um diretório
usermod -aG nomeGrupo:nomeUsuario = colocar user como adm do grupo
grep nomeUser /etc/passwd = ver os negocios do usuario
useradd -s /bin/bash nomeUser = cria usuario add ele no bash
gpasswd -d nomeUser nomeGrupo = remove user do grupo´

permissões

rwx rwx rwx
usuario grupo outros

u - usuario propietario
g - propietario do grupo
o - outros
a - todos

+ adicionar
- remover
= igual

r - read
w - write
x - execute

exemplo:

chmod u+r arquivo.txt
chmod g-w arquivo.txt
chmod u=rx arquivo.txt
cmhod +t arquivo.txt
